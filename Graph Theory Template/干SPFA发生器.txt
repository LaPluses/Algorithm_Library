#include <cstdio>
#include <cstdlib>

int main()
{
	int n = 100000;
	freopen("B1.in", "w", stdout);
	printf("100000 %d\n", n * 2 - 3);
	for (int i = 2; i <= n; ++i)
		printf("%d %d %d\n", 1, i, 1000000000 / i);
	for (int i = 1; i < n - 1; ++i)
		printf("%d %d %d\n", i + 1, i + 2, 1);
	printf("1 100000\n");
	return 0;
}
